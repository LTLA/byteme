121 // Seeing if we fill up the rest of our buffer or not.
122 const std::size_t remaining = buffer_size - my_current;
123 if (number < remaining) {
124 std::copy_n(input, number, my_buffer.data() + my_current);
125 my_current += number;
126 return;
127 }
128
129 std::copy_n(input, remaining, my_buffer.data() + my_current);
130 number -= remaining;
131 input += remaining;
132
133 if (number == 0) {
134 flush_async();

136 my_current = 0;
137 return;
138 }
139
140 flush_sync(my_buffer.data(), buffer_size);
141 my_overall += buffer_size;
142
143 // Directly filling the output array, bypassing our buffer.
144 while (number >= buffer_size) {
145 flush_sync(input, buffer_size);
146 my_overall += buffer_size;
147 input += buffer_size;
148 number -= buffer_size;
149 }
150
151 // Filling the cache and copying the remainder into the output array.
152 std::copy_n(input, number, my_buffer.data());

365 void flush_sync(const Type_* ptr, std::size_t num) {
366 if (my_worker_active) {
367 // If the worker is active, we wait for it to finish and then write from the supplied pointer.
368 // We do not submit a new job, based on the loop in the BufferedWriter::write() overload:
369 //
370 // - We'll probably want to call this flush() overload again.
371 // On subsequent calls, we'll want to directly write from the supplied pointer to the Writer to cut out a copy.
372 // - If we didn't call this overload again, we'd probably call the other refill() overload immediately,
373 // So if we sent a new job to the worker, we'd end up immediately blocking on it to submit the next write.
374 // So we might as well skip that communication overhead and write directly on the main thread.
375 std::unique_lock lck(my_mut);
376 my_cv.wait(lck, [&]() { return my_ready_output; });
377 my_ready_output = false;
378 if (my_thread_err) {
379 std::rethrow_exception(my_thread_err);
380 }
381
382 my_writer->write(reinterpret_cast<const unsigned char*>(ptr), num);
383 my_worker_active = false;
384
385 } else {
386 // If the worker isn't active, we can directly write from the supplied pointer, avoiding some communication overhead.
