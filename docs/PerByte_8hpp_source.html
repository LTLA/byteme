23void skip_zero_buffers(Pointer_& reader, size_t& available) {
24 available = 0;
25 while (reader->load()) {
26 available = reader->available(); // continue collecting bytes if a zero-length buffer is returned without load() returning false.
27 if (available) {
28 break;
29 }
30 }
31
32 // If available == 0 on return, then reader->load() must be false,
33 // and there are no more bytes left in the source.
34}
