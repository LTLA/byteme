145 std::pair<std::size_t, bool> extract(std::size_t number, Type_* output) {
146 const std::size_t original = number;
147
148 // Copying contents of the cached buffer.
149 const std::size_t remaining = my_available - my_current;
150 if (number < remaining) {
151 std::copy_n(my_buffer_ptr + my_current, number, output);
152 my_current += number;
153 return std::make_pair(number, true);
154 }
155
156 std::copy_n(my_buffer_ptr + my_current, remaining, output);
157 output += remaining;
158 number -= remaining;
159
160 // If the available number of bytes is less than the buffer size, the reader is already exhausted.
161 if (my_available < buffer_size) {
162 my_current = my_available;
163 return std::make_pair(original - number, false);
164 }
165
166 // Directly filling the output array, bypassing our buffer.
167 while (number >= buffer_size) {
168 my_overall += my_available;
169 my_available = refill(output);
170
171 output += my_available;
172 number -= my_available;
173 if (my_available < buffer_size) {
174 my_current = my_available;
175 return std::make_pair(original - number, false);
176 }
177 }
178
179 // Filling the cache and copying the remainder into the output array.
180 my_overall += my_available;
181 auto rout = refill();
182 my_buffer_ptr = rout.first;
183 my_available = rout.second;
184
185 const auto to_use = std::min(my_available, number);
186 std::copy_n(my_buffer_ptr, to_use, output);
187 number -= to_use;
188 my_current = to_use;
189
190 // We know that number < buffer_size from the loop condition, so if exhausted == true,
191 // this means that my_available <= number < buffer_size, i.e., the source is exhausted.
192 // Thus, any future advance() would definitely return false.
193 bool exhausted = (to_use == my_available);
194
195 return std::make_pair(original - number, !exhausted);
196 }
